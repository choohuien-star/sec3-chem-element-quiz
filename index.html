<script>

    const API_URL = ""; // PASTE GOOGLE WEB APP URL HERE

    const TEACHER_PASSWORD = "ilovechemistry";


    const questions = [

        { text: "A complex stoichiometry problem appears on the whiteboard. What's your move?", options: [{ text: "Calculate in my head.", element: "Magnesium" }, { text: "Double-check the formula.", element: "Carbon" }, { text: "Listen for hints.", element: "Argon" }, { text: "List out given values.", element: "Copper" }] },

        { text: "How do you feel about practical lab sessions?", options: [{ text: "I love the hands-on part!", element: "Magnesium" }, { text: "Follow procedure perfectly.", element: "Carbon" }, { text: "Observe theory in life.", element: "Argon" }, { text: "Ensure safety/tidiness.", element: "Copper" }] },

        { text: "It's 9:00 PM and you have Chemistry homework due...", options: [{ text: "I finished it during recess.", element: "Magnesium" }, { text: "Logical explanations.", element: "Carbon" }, { text: "Understanding it first.", element: "Argon" }, { text: "Never miss a deadline.", element: "Copper" }] },

        { text: "Your group is stuck on a difficult question. How do you help?", options: [{ text: "Bold new ways.", element: "Magnesium" }, { text: "Organize what we know.", element: "Carbon" }, { text: "Encourage everyone.", element: "Argon" }, { text: "Find a similar example.", element: "Copper" }] },

        { text: "What motivates you most in a Chemistry lesson?", options: [{ text: "The thrill of discovery.", element: "Magnesium" }, { text: "Solid foundation for future.", element: "Carbon" }, { text: "Seeing science explain life.", element: "Argon" }, { text: "Giving my best effort.", element: "Copper" }] },

        { text: "When you make a mistake in a test, how do you react?", options: [{ text: "Check logic immediately.", element: "Magnesium" }, { text: "Rewrite clearly.", element: "Carbon" }, { text: "Reflect on misunderstanding.", element: "Argon" }, { text: "Ask for extra practice.", element: "Copper" }] }

    ];


    const elements = {

        "Magnesium": { title: "Magnesium", subtitle: "The Radiant Catalyst", icon: "🔥", desc: "Bright, energetic, and proactive! You start the reaction." },

        "Carbon": { title: "Carbon", subtitle: "The Structural Backbone", icon: "💎", desc: "Versatile and strong! You provide stability." },

        "Argon": { title: "Argon", subtitle: "The Balanced Observer", icon: "☁️", desc: "Cool-headed and insightful! You bring harmony." },

        "Copper": { title: "Copper", subtitle: "The Reliable Conductor", icon: "⚡", desc: "Efficient and dedicated! You bridge ideas and results." }

    };


    let currentQuestion = 0;

    let scores = { Magnesium: 0, Carbon: 0, Argon: 0, Copper: 0 };

    let studentData = { name: "", class: "", answers: [], result: "" };


    const teacherBtn = document.getElementById('teacher-btn');

    const passwordModal = document.getElementById('password-modal');

    const modalCancel = document.getElementById('modal-cancel');

    const modalSubmit = document.getElementById('modal-submit');

    const passwordInput = document.getElementById('teacher-password-input');


    teacherBtn.onclick = () => passwordModal.classList.remove('hidden');

    modalCancel.onclick = () => { passwordModal.classList.add('hidden'); passwordInput.value = ''; };

    modalSubmit.onclick = () => {

        if (passwordInput.value === TEACHER_PASSWORD) {

            passwordModal.classList.add('hidden');

            document.getElementById('screen-welcome').classList.add('hidden');

            document.getElementById('screen-teacher').classList.remove('hidden');

            document.getElementById('main-container').style.maxWidth = "950px";

            loadDashboardData();

        } else { 

            alert("Incorrect code."); 

            passwordInput.value = ''; 

        }

    };


    function startQuiz() {

        const n = document.getElementById('student-name').value.trim();

        const c = document.getElementById('student-class').value.trim();

        if (!n || !c) return alert("Please enter your name and class!");

        studentData.name = n; 

        studentData.class = c;

        document.getElementById('screen-welcome').classList.add('hidden');

        document.getElementById('screen-quiz').classList.remove('hidden');

        showQuestion();

    }


    function showQuestion() {

        const q = questions[currentQuestion];

        document.getElementById('q-text').innerText = q.text;

        const opts = document.getElementById('q-options');

        opts.innerHTML = '';

        q.options.forEach(o => {

            const b = document.createElement('div');

            b.className = 'option-btn'; 

            b.innerText = o.text;

            b.onclick = () => { 

                scores[o.element]++; 

                studentData.answers.push(o.text); 

                next(); 

            };

            opts.appendChild(b);

        });

        document.getElementById('progress-fill').style.width = `${(currentQuestion / questions.length) * 100}%`;

    }


    function next() {

        if (++currentQuestion < questions.length) {

            showQuestion();

        } else {

            let res = Object.keys(scores).reduce((a, b) => scores[a] > scores[b] ? a : b);

            studentData.result = res;

            showResult(res);

            if (API_URL) {

                fetch(API_URL, { 

                    method: 'POST', 

                    mode: 'no-cors', 

                    body: JSON.stringify(studentData) 

                });

            }

        }

    }


    function showResult(el) {

        document.getElementById('screen-quiz').classList.add('hidden');

        document.getElementById('screen-result').classList.remove('hidden');

        const d = elements[el];

        document.getElementById('result-icon').innerText = d.icon;

        document.getElementById('result-element').innerText = d.title;

        document.getElementById('result-subtitle').innerText = d.subtitle;

        document.getElementById('result-desc').innerText = d.desc;

    }


    let allResponses = [];

    async function loadDashboardData() {

        if (!API_URL) return;

        try {

            const res = await fetch(API_URL);

            allResponses = await res.json();

            const tbody = document.getElementById('dashboard-body');

            tbody.innerHTML = '';

            allResponses.forEach(r => {

                const tr = document.createElement('tr');

                tr.innerHTML = `<td>${r.name}</td><td>${r.class}</td><td><span class="tag tag-${r.result.toLowerCase()}">${r.result}</span></td>` + 

                               r.answers.map(a => `<td>${a}</td>`).join('');

                tbody.appendChild(tr);

            });

        } catch(e) { 

            console.log("Database fetch failed. Ensure your URL is correct and deployed."); 

        }

    }


    function downloadCSV() {

        if (allResponses.length === 0) return alert("No data to export!");

        let csv = "Name,Class,Result,Q1,Q2,Q3,Q4,Q5,Q6\n" + 

                  allResponses.map(r => `"${r.name}","${r.class}","${r.result}","${r.answers.join('","')}"`).join("\n");

        const blob = new Blob([csv], { type: 'text/csv' });

        const a = document.createElement('a');

        a.href = URL.createObjectURL(blob);

        a.download = `Student_Elements_Results.csv`; 

        a.click();

    }
